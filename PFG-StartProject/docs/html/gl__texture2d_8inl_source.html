150  inline GLuint createTexture2D(std::string const & Filename)
151  {
152  gli::texture2D Texture = gli::load(Filename);
153  if(Texture.empty())
154  return 0;
155 
156  detail::texture_desc TextureDesc = detail::gli2ogl_cast(Texture.format());
157 
158  GLint Alignment = 0;
159  glGetIntegerv(GL_UNPACK_ALIGNMENT, &Alignment);
160 
161  glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
162 
163  GLuint Name = 0;
164  glGenTextures(1, &Name);
165  glBindTexture(GL_TEXTURE_2D, Name);
166  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, Texture.levels() > 1 ? GL_NEAREST_MIPMAP_NEAREST : GL_NEAREST);
167  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
168 
169  if(size(Texture, BIT_PER_PIXEL) == size(Texture, BLOCK_SIZE) << 3)
170  {
171  for(gli::texture2D::level_type Level = 0; Level < Texture.levels(); ++Level)
172  {
173  glTexImage2D(
174  GL_TEXTURE_2D,
175  GLint(Level),
176  TextureDesc.InternalFormat,
177  GLsizei(Texture[Level].dimensions().x),
178  GLsizei(Texture[Level].dimensions().y),
179  0,
180  TextureDesc.ExternalFormatRev,
181  TextureDesc.Type,
182  Texture[Level].data());
183  }
184  }
185  else
186  {
187  for(gli::texture2D::level_type Level = 0; Level < Texture.levels(); ++Level)
188  {
189  glCompressedTexImage2D(
190  GL_TEXTURE_2D,
191  GLint(Level),
192  TextureDesc.InternalFormat,
193  GLsizei(Texture[Level].dimensions().x),
194  GLsizei(Texture[Level].dimensions().y),
195  0,
196  GLsizei(Texture[Level].capacity()),
197  Texture[Level].data());
198  }
199  }
200 
201  glBindTexture(GL_TEXTURE_2D, 0);
202 
203  glPixelStorei(GL_UNPACK_ALIGNMENT, Alignment);
204 
205  return Name;
206  }
