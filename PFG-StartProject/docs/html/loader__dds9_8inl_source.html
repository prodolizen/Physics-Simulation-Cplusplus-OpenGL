395  inline texture2D loadDDS9
396  (
397  std::string const & Filename
398  )
399  {
400  std::ifstream FileIn(Filename.c_str(), std::ios::in | std::ios::binary);
401  if(FileIn.fail())
402  return texture2D();
403 
404  detail::ddsHeader SurfaceDesc;
405  char Magic[4];
406 
407  //* Read magic number and check if valid .dds file
408  FileIn.read((char*)&Magic, sizeof(Magic));
409 
410  assert(strncmp(Magic, "DDS ", 4) == 0);
411 
412  // Get the surface descriptor
413  FileIn.read((char*)&SurfaceDesc, sizeof(SurfaceDesc));
414 
415  std::size_t Width = SurfaceDesc.width;
416  std::size_t Height = SurfaceDesc.height;
417 
418  //std::size_t Levels = glm::max(glm::highestBit(Width), glm::highestBit(Height));
419 
420  detail::DDLoader Loader;
421  if(SurfaceDesc.format.flags & detail::GLI_DDPF_FOURCC)
422  {
423  switch(SurfaceDesc.format.fourCC)
424  {
425  case detail::GLI_FOURCC_DX10:
426  assert(0);
427  break;
428  case detail::GLI_FOURCC_DXT1:
429  Loader.BlockSize = 8;
430  Loader.Format = DXT1;
431  break;
432  case detail::GLI_FOURCC_DXT3:
433  Loader.BlockSize = 16;
434  Loader.Format = DXT3;
435  break;
436  case detail::GLI_FOURCC_DXT5:
437  Loader.BlockSize = 16;
438  Loader.Format = DXT5;
439  break;
440  case detail::GLI_FOURCC_R16F:
441  Loader.BlockSize = 2;
442  Loader.Format = R16F;
443  break;
444  case detail::GLI_FOURCC_G16R16F:
445  Loader.BlockSize = 4;
446  Loader.Format = RG16F;
447  break;
448  case detail::GLI_FOURCC_A16B16G16R16F:
449  Loader.BlockSize = 8;
450  Loader.Format = RGBA16F;
451  break;
452  case detail::GLI_FOURCC_R32F:
453  Loader.BlockSize = 4;
454  Loader.Format = R32F;
455  break;
456  case detail::GLI_FOURCC_G32R32F:
457  Loader.BlockSize = 8;
458  Loader.Format = RG32F;
459  break;
460  case detail::GLI_FOURCC_A32B32G32R32F:
461  Loader.BlockSize = 16;
462  Loader.Format = RGBA32F;
463  break;
464 
465  default:
466  assert(0);
467  return texture2D();
468  }
469  }
470  else if(SurfaceDesc.format.flags & detail::GLI_DDPF_RGB)
471  {
472  switch(SurfaceDesc.format.bpp)
473  {
474  case 8:
475  Loader.BlockSize = 2;
476  Loader.Format = R8U;
477  break;
478  case 16:
479  Loader.BlockSize = 2;
480  Loader.Format = RG8U;
481  break;
482  case 24:
483  Loader.BlockSize = 3;
484  Loader.Format = RGB8U;
485  break;
486  case 32:
487  Loader.BlockSize = 4;
488  Loader.Format = RGBA8U;
489  break;
490  }
491  }
492  else
493  {
494 
495  }
496 
497  gli::format Format = Loader.Format;
498 
499  std::streamoff Curr = FileIn.tellg();
500  FileIn.seekg(0, std::ios_base::end);
501  std::streamoff End = FileIn.tellg();
502  FileIn.seekg(Curr, std::ios_base::beg);
503 
504  std::vector<glm::byte> Data(std::size_t(End - Curr), 0);
505  std::size_t Offset = 0;
506 
507  FileIn.read((char*)&Data[0], std::streamsize(Data.size()));
508 
509  //image Image(glm::min(MipMapCount, Levels));//SurfaceDesc.mipMapLevels);
510  std::size_t MipMapCount = (SurfaceDesc.flags & detail::GLI_DDSD_MIPMAPCOUNT) ? SurfaceDesc.mipMapLevels : 1;
511  //if(Loader.Format == DXT1 || Loader.Format == DXT3 || Loader.Format == DXT5)
512  // MipMapCount -= 2;
513  texture2D Image(MipMapCount);
514  for(std::size_t Level = 0; Level < Image.levels() && (Width || Height); ++Level)
515  {
516  Width = glm::max(std::size_t(Width), std::size_t(1));
517  Height = glm::max(std::size_t(Height), std::size_t(1));
518 
519  std::size_t MipmapSize = 0;
520  if(Loader.Format == DXT1 || Loader.Format == DXT3 || Loader.Format == DXT5)
521  MipmapSize = ((Width + 3) >> 2) * ((Height + 3) >> 2) * Loader.BlockSize;
522  else
523  MipmapSize = Width * Height * Loader.BlockSize;
524  std::vector<glm::byte> MipmapData(MipmapSize, 0);
525 
526  memcpy(&MipmapData[0], &Data[0] + Offset, MipmapSize);
527 
528  image2D::dimensions_type Dimensions(Width, Height);
529  Image[Level] = image2D(Dimensions, Format, MipmapData);
530 
531  Offset += MipmapSize;
532  Width >>= 1;
533  Height >>= 1;
534  }
535 
536  return Image;
537  }
